Lets simulate something..

I have functions

sim_mean_sd = function(sample_size, mu = 3, sigma = 4) {
  
  sim_data = 
  tibble(
    x = rnorm(n = sample_size, mean = mu, sd = sigma)
  )

  sim_data %>% 
    summarise(
      mean = mean(x),
      sd = sd(x)
    )

}

I can “simulate” by running this line.

sim_mean_sd(30)

sim_results = 
  rerun(5000, sim_mean_sd(sample_size = 30)) %>%

sim_results = 
  tibble(
    sample_size = c(30, 60, 120, 240)
  ) %>% 
  mutate(
    output_list = map(.x = sample_size, ~ rerun(10, sim_mean_sd(.x))),
    estimate_df = map(output_list, bind_rows)
    ) %>% 
    select(-output_list) %>% 
    unnest(estimate_df)
